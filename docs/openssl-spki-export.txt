语法:
   string openssl_spki_export    ( string &$spkac   )
说明:
ExportsPEMformattedpublickeyfromencodedsignedpublickeyandchallenge
参数:
  参数           spkac                   Expects a valid signed public key and challenge                   
返回值:
  返回值     Returns the associated PEM formatted public key or NULL on failure.   
示例:

      Extracts the associated PEM formatted public key or NULL on failure.
   
<?php
$pkey = openssl_pkey_new('secret password');
$spkac = openssl_spki_new($pkey, 'challenge string');
$pubKey = openssl_spki_export(preg_replace('/SPKAC=/', '', $spkac));
if ($pubKey) {
    echo $pubKey;
}
?>

      Extracts the associated PEM formatted public key issued from the <keygen> element
   
<?php
$spkac = openssl_spki_export(preg_replace('/SPKAC=/', '', $_POST['spkac']));
if ($spkac != NULL) {
    echo $spkac;
} else {
    echo "Extraction of pub key failed";
}
?>
